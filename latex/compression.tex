\section{RLE}
\label{sec:rle}

\begin{algorithm}
  \caption{Decoding a RLE encoded file.}
  \begin{algorithmic}[1]
    \Let{$length$}{$-1$} \Comment{The number of bytes to uncomppress.}

    \Let{$b$}{ \Call{Read}{}} \Comment{Reads the next byte.}
    \While{$b\not=EOF$} \Comment{While the end the file hasn't yet
      been reached.}
      \If{$length = -1$}
        \Let{$length$}{$b$}
      \Else
        \Repeat{$length$}
          \State \Call{Write}{$b$}
        \EndRepeat
        \Let{$length$}{$-1$}
      \EndIf
      \Let{$b$}{\Call{Read}{}}
    \EndWhile
  \end{algorithmic}
\end{algorithm}
